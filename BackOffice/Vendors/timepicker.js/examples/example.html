        var timepicker = new TimePicker(['time', 'link'], {
          theme: 'dark', // 'blue-grey'
          lang: 'en'
        });

        timepicker.on('change', function(evt) {
          var value = (evt.hour || '00') + ':' + (evt.minute || '00');

          if (evt.element.id == 'link') {
            time2.value = value;
          } else {
            evt.element.value = value;
          }
        });
